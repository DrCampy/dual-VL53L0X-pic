I2C for tof uses I2C1:
Configuration registers

SSP1STAT = 0b0000 0000 1000 0000 (SSP1STAT.SMP=1 /*If using standard speed mode*/; SSP1STAT.CKE=0)

SSP1CON1 = 0b0000 0000 xx1x 1000 (SSP1CON1.SSPEN=1; SSP1CON1.SSPM=1000)

SSP1CON2 = 0b0000 0000 0xxx xxxx (SSP1CON2.GCEN=0;)

SSP1CON3 = 0b0000 0000 x00x xxxx (SSP1CON3.PCIE=0; SSP1CON3.SCIE=0;)
